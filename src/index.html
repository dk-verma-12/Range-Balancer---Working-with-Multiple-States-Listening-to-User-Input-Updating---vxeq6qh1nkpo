const rangeA = document.getElementById('range-a'); const rangeB = document.getElementById('range-b'); const maxSum = document.getElementById('max-sum'); const rangeAValue = document.getElementById('range-a-value'); const rangeBValue = document.getElementById('range-b-value'); const sum = document.getElementById('sum'); function updateRanges() { const max = Number(maxSum.value); const rangeAVal = Number(rangeA.value); const rangeBVal = Number(rangeB.value); const total = rangeAVal + rangeBVal; if (total > max) { // Adjust range values to stay within max sum const newRangeAVal = Math.min(rangeAVal, max - rangeBVal); const newRangeBVal = Math.min(rangeBVal, max - rangeAVal); // Update range values rangeA.value = newRangeAVal; rangeB.value = newRangeBVal; // Update range value displays rangeAValue.textContent = newRangeAVal; rangeBValue.textContent = newRangeBVal; } else { // Update range value displays rangeAValue.textContent = rangeAVal; rangeBValue.textContent = rangeBVal; } // Update total display sum.textContent = total; } rangeA.addEventListener('input', updateRanges); rangeB.addEventListener('input', updateRanges); maxSum.addEventListener('input', updateRanges); updateRanges();
